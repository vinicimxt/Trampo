create database Xamou;
use Xamou;


CREATE TABLE Usuarios (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100),
    Email VARCHAR(100) UNIQUE,
    Senha VARCHAR(100),
    Tipo VARCHAR(20) -- Cliente ou Profissional
);

CREATE TABLE Clientes (
    Id INT PRIMARY KEY IDENTITY,
    UsuarioId INT,
    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id)
);

CREATE TABLE Profissionais (
    Id INT PRIMARY KEY IDENTITY,
    Documento VARCHAR(20),
    TipoDocumento VARCHAR(10),
    UsuarioId INT,
    TipoServico VARCHAR(100),
    Descricao VARCHAR(255),
    Atendimento VARCHAR(50), -- domicilio ou local
    RaioAtendimento INT,
    Latitude FLOAT,
    Longitude FLOAT,
    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id)
);

CREATE TABLE Disponibilidade (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT,
    DiaSemana INT, -- 1 a 7
    HoraInicio TIME,
    HoraFim TIME,
    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id)
);

CREATE TABLE Agendamentos (
    Id INT PRIMARY KEY IDENTITY,
    ClienteId INT,
    ProfissionalId INT,
    Data DATE,
    Hora TIME,
    Status VARCHAR(20), -- Pendente, Confirmado, Cancelado
    FOREIGN KEY (ClienteId) REFERENCES Clientes(Id),
    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id)
);
