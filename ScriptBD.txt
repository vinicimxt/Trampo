-- =============================
-- TABELAS
-- =============================

CREATE TABLE Usuarios (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Senha VARCHAR(255) NOT NULL,
    Tipo VARCHAR(20) NOT NULL,
    Telefone VARCHAR(20) NULL
);

CREATE TABLE Clientes (
    Id INT PRIMARY KEY IDENTITY,
    UsuarioId INT NOT NULL UNIQUE,
    BloqueadoAte DATETIME NULL,

    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id) ON DELETE CASCADE
);

CREATE TABLE Profissionais (
    Id INT PRIMARY KEY IDENTITY,
    UsuarioId INT NOT NULL UNIQUE,

    TipoDocumento VARCHAR(10), -- CPF ou CNPJ
    Documento VARCHAR(20),

    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id) ON DELETE CASCADE
);

--  UNIQUE apenas para CPF
CREATE UNIQUE INDEX UX_Profissional_CPF
ON Profissionais(Documento)
WHERE TipoDocumento = 'CPF';

-- =============================

CREATE TABLE Categorias (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100) NOT NULL
);

CREATE TABLE Subcategorias (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100) NOT NULL,
    CategoriaId INT NOT NULL,

    FOREIGN KEY (CategoriaId) REFERENCES Categorias(Id) ON DELETE CASCADE
);

CREATE TABLE Locais (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,
    Nome VARCHAR(100),
    Endereco VARCHAR(255) NOT NULL,

    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id) ON DELETE CASCADE
);

CREATE TABLE Servicos (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,
    SubcategoriaId INT NOT NULL,
    LocalId INT NULL,

    Nome NVARCHAR(100) NOT NULL,
    Descricao NVARCHAR(255),
    Contato VARCHAR(100),
    Atendimento VARCHAR(50),
    LinkOnline VARCHAR(255) NULL,

    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id) ON DELETE CASCADE,
    FOREIGN KEY (SubcategoriaId) REFERENCES Subcategorias(Id),
    FOREIGN KEY (LocalId) REFERENCES Locais(Id)
);

CREATE TABLE Disponibilidade (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,

    DiaSemana INT NOT NULL,
    HoraInicio TIME NOT NULL,
    HoraFim TIME NOT NULL,

    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id) ON DELETE CASCADE
);

CREATE TABLE Agendamentos (
    Id INT PRIMARY KEY IDENTITY,

    ClienteId INT NOT NULL,
    ProfissionalId INT NOT NULL,
    ServicoId INT NOT NULL,
    LocalId INT NULL,

    Data DATE NOT NULL,
    Hora TIME NOT NULL,

    Status VARCHAR(50) NOT NULL,
    Descricao VARCHAR(255),
    DataCancelamento DATETIME NULL,
    EnderecoCliente VARCHAR(255) NULL,

    ConfirmadoProfissional BIT DEFAULT 0,
    FinalizadoProfissional BIT DEFAULT 0,
    ConfirmadoCliente BIT DEFAULT 0,

    FOREIGN KEY (ClienteId) REFERENCES Clientes(Id),
    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id),
    FOREIGN KEY (ServicoId) REFERENCES Servicos(Id) ON DELETE CASCADE,
    FOREIGN KEY (LocalId) REFERENCES Locais(Id)
);

CREATE TABLE Avaliacoes (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,
    UsuarioId INT NOT NULL,
    AgendamentoId INT NOT NULL,
    Nota INT NOT NULL,
    Comentario NVARCHAR(500),
    Data DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (ProfissionalId) REFERENCES Profissionais(Id),
    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id),
    FOREIGN KEY (AgendamentoId) REFERENCES Agendamentos(Id)
);

--  evita avaliar 2x o mesmo serviço
CREATE UNIQUE INDEX UX_Avaliacao_Agendamento
ON Avaliacoes(AgendamentoId);
