-- 🔥 LIMPA BANCO (CUIDADO EM PRODUÇÃO)
DROP TABLE IF EXISTS Agendamentos;
DROP TABLE IF EXISTS Disponibilidade;
DROP TABLE IF EXISTS Servicos;
DROP TABLE IF EXISTS Clientes;
DROP TABLE IF EXISTS Profissionais;
DROP TABLE IF EXISTS Usuarios;

-- 👤 USUÁRIOS
CREATE TABLE Usuarios (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Senha VARCHAR(255) NOT NULL,
    Tipo VARCHAR(20) NOT NULL -- Cliente ou Profissional
);

-- 👤 CLIENTES
CREATE TABLE Clientes (
    Id INT PRIMARY KEY IDENTITY,
    UsuarioId INT NOT NULL,
    BloqueadoAte DATETIME NULL,

    FOREIGN KEY (UsuarioId) 
        REFERENCES Usuarios(Id)
        ON DELETE CASCADE
);

-- 🧑‍🔧 PROFISSIONAIS (AGORA SÓ PERFIL)
CREATE TABLE Profissionais (
    Id INT PRIMARY KEY IDENTITY,
    UsuarioId INT NOT NULL,
    TipoDocumento VARCHAR(10),
    Documento VARCHAR(20),

    FOREIGN KEY (UsuarioId) 
        REFERENCES Usuarios(Id)
        ON DELETE CASCADE
);

-- 🛠 SERVIÇOS (AGORA É O CORAÇÃO)
CREATE TABLE Servicos (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,

    Nome NVARCHAR(100) NOT NULL,
    Descricao NVARCHAR(255),
    Contato VARCHAR(100),

    Atendimento VARCHAR(50), -- Domicilio / Local / Online
    RaioAtendimento INT NULL,

    FOREIGN KEY (ProfissionalId) 
        REFERENCES Profissionais(Id)
        ON DELETE CASCADE
);

CREATE TABLE Categorias (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100)
);


CREATE TABLE Subcategorias (
    Id INT PRIMARY KEY IDENTITY,
    Nome VARCHAR(100),
    CategoriaId INT,
    FOREIGN KEY (CategoriaId) REFERENCES Categorias(Id)
);


-- 📅 DISPONIBILIDADE
CREATE TABLE Disponibilidade (
    Id INT PRIMARY KEY IDENTITY,
    ProfissionalId INT NOT NULL,

    DiaSemana INT NOT NULL, -- 1 a 7
    HoraInicio TIME NOT NULL,
    HoraFim TIME NOT NULL,

    FOREIGN KEY (ProfissionalId) 
        REFERENCES Profissionais(Id)
        ON DELETE CASCADE
);

-- 📌 AGENDAMENTOS (AGORA LIGADO AO SERVIÇO)
CREATE TABLE Agendamentos (
    Id INT PRIMARY KEY IDENTITY,

    ClienteId INT NOT NULL,
    ProfissionalId INT NOT NULL,
    ServicoId INT NOT NULL,

    Data DATE NOT NULL,
    Hora TIME NOT NULL,

    Status VARCHAR(50) NOT NULL, 
    -- Pendente, Confirmado, CanceladoCliente, CanceladoProfissional, Finalizado

    Descricao VARCHAR(255),
    DataCancelamento DATETIME NULL,

    FOREIGN KEY (ClienteId) 
        REFERENCES Clientes(Id),

    FOREIGN KEY (ProfissionalId) 
        REFERENCES Profissionais(Id),

    FOREIGN KEY (ServicoId) 
        REFERENCES Servicos(Id)
);
